precision highp float; uniform vec2 u_resolution; uniform float u_time; const float PI = 3.14159265359; const int MAX_ITERATIONS = 64; const float EPSILON = 0.0001; vec2 complex_mul(vec2 a, vec2 b) { return vec2(a.x*b.x - a.y*b.y, a.x*b.y + a.y*b.x); } vec2 complex_div(vec2 a, vec2 b) { float denom = dot(b, b); return vec2(dot(a, b), a.y*b.x - a.x*b.y) / denom; } vec2 mobius_transform(vec2 z, vec2 a, vec2 b, vec2 c, vec2 d) { vec2 num = complex_mul(a, z) + b; vec2 den = complex_mul(c, z) + d; return complex_div(num, den); } float hyperbolic_distance(vec2 z, vec2 w) { vec2 num = z - w; vec2 den = vec2(1.0) - complex_mul(w, vec2(z.x, -z.y)); float norm = length(complex_div(num, den)); return log((1.0 + norm) / (1.0 - norm)); } // Function to create branching structure float branch(vec2 z, float scale) { float d = 0.0; for (int i = 0; i < 5; i++) { float angle = float(i) * 2.0 * PI / 5.0; vec2 offset = 0.5 * vec2(cos(angle), sin(angle)); d += 0.1 / (0.1 + hyperbolic_distance(z, offset * scale)); } return d; } // Function to create five-fold symmetry pattern float fivefold_symmetry(vec2 z) { float angle = atan(z.y, z.x); return cos(5.0 * angle) * 0.5 + 0.5; } void color_function(vec2 z, out vec3 color) { float dist = hyperbolic_distance(z, vec2(0.0)); // Base color based on distance from center (spherical tendency) color = 0.5 + 0.5*cos(2.0*PI*vec3(dist*0.1) + vec3(0,2,4)); // Add branching structure (hyperbolic tendency) float branch_factor = branch(z, 1.0) + 0.5 * branch(z, 0.5) + 0.25 * branch(z, 0.25); color += vec3(0.2, 0.1, 0.0) * branch_factor; // Add five-fold symmetry pattern float symmetry = fivefold_symmetry(z); color += vec3(0.1, 0.2, 0.3) * symmetry; // Dynamic pattern to show curvature for (int i = 0; i < MAX_ITERATIONS; i++) { z = mobius_transform(z, vec2(cos(u_time*0.1), sin(u_time*0.1)), vec2(0.0), vec2(0.0), vec2(1.0)); if (length(z) > 1.0 - EPSILON) break; color += 0.01 * vec3(0.5 + 0.5*sin(float(i)*0.1 + u_time)); } // Emphasize the edge of the disk to show the "infinity" of hyperbolic space color *= 1.0 - smoothstep(0.9, 1.0, length(z)); color = clamp(color, 0.0, 1.0); } void main() { vec2 uv = (gl_FragCoord.xy - 0.5*u_resolution.xy) / min(u_resolution.x, u_resolution.y); float r = length(uv); if (r < 1.0) { vec3 color; color_function(uv, color); gl_FragColor = vec4(color, 1.0); } else { gl_FragColor = vec4(0.0, 0.0, 0.0, 1.0); } }
